1. 시작하기

필수 조건

  • .NET 6.0 이상
  • Visual Studio 2022 또는 VS Code
  • 키움증권 계정 및 API 접근 권한
  • 키움 OpenAPI에서 발급받은 앱 키와 시크릿 키

설치

NuGet 패키지를 설치하세요:

dotnet add package KiwoomRestApi.Net
팁: NuGet 패키지 관리자에서도 설치할 수 있습니다: Install-Package KiwoomRestApi.Net

기본 프로젝트 설정

using KiwoomRestApi.Net.Clients;
using KiwoomRestApi.Net.Enums.StockInfo;

var appKey = "YOUR_APP_KEY";
var secretKey = "YOUR_SECRET_KEY";
var isMock = true; // 실거래의 경우 false로 설정

// 클라이언트 생성
var client = await KiwoomRestApiClient.CreateAsync(appKey, secretKey, isMock);

2. 기본 사용법

주식 정보 가져오기

삼성전자(005930)의 기본 주식 정보를 가져와 보겠습니다:

// 삼성전자(005930) 주식 정보 가져오기
var stockCode = "005930";
var date = DateTime.Today;

var stockInfo = await client.StockInfo.GetStockInfoAsync(
    stockCode: stockCode,
    date: date,
    marginLoanType: KiwoomStockInfoMarginLoanType.Loan
);

if (stockInfo.IsSuccess)
{
    Console.WriteLine($"주식명: {stockInfo.Data?.StockName}");
    Console.WriteLine($"현재가: {stockInfo.Data?.CurrentPrice:N0} KRW");
    Console.WriteLine($"변동률: {stockInfo.Data?.ChangeRate:F2}%");
    Console.WriteLine($"거래량: {stockInfo.Data?.AccumulatedTradingVolume:N0}");
}
else
{
    Console.WriteLine($"오류: {stockInfo.ErrorMessage}");
}

응답 이해하기

모든 API 호출은 KiwoomRestApiResponse<T> 객체를 반환합니다:

public class KiwoomRestApiResponse<T>
{
    public bool IsSuccess { get; set; }
    public T? Data { get; set; }
    public string? ErrorMessage { get; set; }
    public string ApiId { get; set; }
    public bool ContYn { get; set; }  // 페이지네이션을 위한 연속 플래그
    public string NextKey { get; set; } // 다음 페이지 키
}

3. 계좌 관리

계좌 잔고 가져오기

// 평가 잔고 가져오기
var balances = await client.Account.GetEvaluationBalancesAsync(

    }
}

일별 상태 가져오기

var dailyStatus = await client.Account.GetDailyStatusAsync();

5. 거래 작업

중요: 거래 작업은 실제 API 키가 필요하며 주의해서 사용해야 합니다. 테스트를 위해 모의투자 모드로 시작하세요.

주문 체결

// 매수 주문 예제
var buyOrder = await client.Order.PlaceOrderAsync(
    orderType: KiwoomOrderType.Buy,
    exchangeType: KiwoomOrderDomesticStockExchangeType.Krx,
    stockCode: "005930",
    quantity: 10,
    transactionType: KiwoomOrderTransactionType.Market
);

if (buyOrder.IsSuccess)
{
    Console.WriteLine($"주문 성공. 주문번호: {buyOrder.Data?.OrderNumber}");
    Console.WriteLine($"주문 상태: {buyOrder.Data?.OrderStatus}");
}
else
{
    Console.WriteLine($"주문 실패: {buyOrder.ErrorMessage}");
}

// 지정가 주문 예제
var limitOrder = await client.Order.PlaceOrderAsync(
    orderType: KiwoomOrderType.Buy,
    exchangeType: KiwoomOrderDomesticStockExchangeType.Krx,
    stockCode: "005930",
    quantity: 10,
    transactionType: KiwoomOrderTransactionType.Limit,
    price: 75000  // 지정가
);

7. 실시간 데이터

WebSocket 클라이언트 설정

// 소켓 클라이언트 생성
var socketClient = await KiwoomSocketClient.CreateAsync(client.Token, isMock: true);

// 실시간 주식 체결 데이터 구독
await socketClient.WebSocket.SubscribeAsync(
    serviceNames: [KiwoomWebSocketServiceName.StockExecution],
    stockCodes: ["005930", "000660", "035420"],
    fidList: "0;10;11;12"  // 현재가, 변동률, 거래량 등
);

실시간 이벤트 처리

// 이벤트 핸들러 설정
socketClient.OnRealtimeStockTradeReceived += (realtimeData) =>
{
    foreach (var data in realtimeData)
    {
        Console.WriteLine($"[{data.StockCode}] {data.Values.CurrentPrice} " +
                         $"({data.Values.ChangeRate:F2}%) " +
                         $"거래량: {data.Values.AccumulatedVolume:N0}");
    }
};

socketClient.OnRealtimeOrderTradeReceived += (orderData) =>
{
    foreach (var order in orderData)
    {
        Console.WriteLine($"주문 업데이트: {order.Values.OrderNumber} - {order.Values.OrderStatus}");
        Console.WriteLine($"주식: {order.Values.StockCode}, 수량: {order.Values.OrderQuantity}");
    }
};

socketClient.OnRealtimeBalanceReceived += (balanceData) =>
{
    foreach (var balance in balanceData)
    {
        Console.WriteLine($"잔고 업데이트: {balance.Values.StockName}");
        Console.WriteLine($"보유 수량: {balance.Values.HoldingQuantity}");
        Console.WriteLine($"평가 금액: {balance.Values.EvaluationAmount:N0}");
    }
};

10. 모범 사례

연결 관리

using var client = await KiwoomRestApiClient.CreateAsync(appKey, secretKey, isMock);
using var socketClient = await KiwoomSocketClient.CreateAsync(client.Token, isMock);

// 또는 직접 해제
try
{
    // API 호출
}
finally
{
    client?.Dispose();
    socketClient?.Dispose();
}

속도 제한

// 요청 간 지연 구현
public class RateLimitedKiwoomClient
{
    private readonly KiwoomRestApiClient _client;
    private readonly SemaphoreSlim _semaphore;
    private readonly TimeSpan _delay;

    public RateLimitedKiwoomClient(KiwoomRestApiClient client, int requestsPerSecond = 10)
    {
        _client = client;
        _semaphore = new SemaphoreSlim(1);
        _delay = TimeSpan.FromMilliseconds(1000.0 / requestsPerSecond);
    }

    public async Task<KiwoomRestApiResponse<T>> ExecuteAsync<T>(Func<Task<KiwoomRestApiResponse<T>>> apiCall)
    {
        await _semaphore.WaitAsync();
        try
        {
            var result = await apiCall();
            await Task.Delay(_delay);
            return result;
        }
        finally
        {
            _semaphore.Release();
        }
    }
}
